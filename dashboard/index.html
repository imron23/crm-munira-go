Kode Inject — Cara LP Memuat Form 
// Di LP2, tambahkan script ini sebelum </body>:
<div id="dynamic-form-container"></div>
<script>
(async function() {
  const res = await fetch('/api/pages/lp-2-long/config');
  const cfg = await res.json();
  if (cfg.linked_form_id) {
    const fRes = await fetch('/api/forms/' + cfg.linked_form_id);
    const fData = await fRes.json();
    window.__MUNIRA_FORM__ = fData.data; // inject ke LP
    injectForm(fData.data); // fungsi render form di LP
  }
})();
</script>
